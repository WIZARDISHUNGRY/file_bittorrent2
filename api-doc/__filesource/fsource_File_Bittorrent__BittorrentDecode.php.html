  111.         /**
    112. 

  112.         * Decode .torrent file and accumulate information
    113. 

  113.         *

    115. 

  115.         * @return bool     Raise PEAR_Error on error
    116.

    117. 

  117.         function decodeFile($file)
    118. 

  118.         {
    119. 

  119.             // Check file
    120. 

  120.             if (!is_file($file)) {
    121. 

  121.                 PEAR::raiseError('File_Bittorrent_Decode::decode() - Not a file.', null, null, "Given filename '$file' is not a valid file.");
    122. 

  122.                 return false;
    123. 

  123.             }
    124. 

  124.             
    125. 

  125.             // Reset public attributes
    126. 

  126.             $this->name       = '';
    127. 

  127.             $this->filename   = '';
    128. 

  128.             $this->comment    = '';
    129. 

  129.             $this->date       = 0;
    130. 

  130.             $this->files      = array();
    131. 

  131.             $this->size       = 0;
    132. 

  132.             $this->created_by = '';
    133. 

  133.             $this->_position  = 0;
    134. 

  134.  
    135. 

  135.             // Decode .torrent  
    136. 

  136.             $this->_source = file_get_contents($file);
    137. 

  137.             $decoded = $this->_bdecode();
    138. 

  138.             
    139. 

  139.             // Pull information form decoded data
    140. 

  140.             $this->filename = basename($file);
    141. 

  141.             // Name of the Torrent - statet by the Torrent's author
    142. 

  142.             $this->name     = $decoded['info']['name'];
    143. 

  143.             // Authors may add comments to a Torrent
    144. 

  144.             if (isset($decoded['comment'])) {
    145. 

  145.                 $this->comment = $decoded['comment'];
    146. 

  146.             }
    147. 

  147.             // Creation date of the Torrent as unix timestamp
    148. 

  148.             if (isset($decoded['creation date'])) {
    149. 

  149.                 $this->date = $decoded['creation date'];
    150. 

  150.             }
    151. 

  151.             // This contains the signature of the application used to create the Torrent
    152. 

  152.             if (isset($decoded['created by'])) {
    153. 

  153.                 $this->created_by = $decoded['created by'];
    154. 

  154.             }
    155. 

  155.             // There is sometimes an array listing all files 
    156. 

  156.             // in the Torrent with their individual filesize
    157. 

  157.             if (isset($decoded['info']['files']) and is_array($decoded['info']['files'])) {
    158. 

  158.                 foreach ($decoded['info']['files'] as $file) {
    159. 

  159.                     $this->size += $file['length'];
    160. 

  160.                     $this->files[] = array(
    161. 

  161.                         'filename' => $file['path'][0],
    162. 

  162.                         'size'     => $file['length'],
    163. 

  163.                     );
    164. 

  164.                 }
    165. 

  165.             }
    166. 

  166.             // This contains the total length of the files in
    167. 

  167.             // the Torrent after the download is completed
    168. 

  168.             // in Bytes
    169. 

  169.             if (isset($decoded['info']['length']) and $this->size == 0) {
    170. 

  170.                 $this->size = $decoded['info']['length'];
    171. 

  171.             }
    172. 

  172.             
    173. 

  173.             // Currently, I'm not sure how to determine an error
    174. 

  174.             // Just try to fetch the info from the decoded data
    175. 

  175.             // and return true
    176. 

  176.             return true;
    177. 

  177.         }

    199. 

  199.         * Decode a BEncoded String
    200. 

  200.         *
    201. 

  201.         * @access private
    202. 

  202.         * @return array    Returns an array containing the representation of the data in the BEncoded string

    204. 

  204.         function _bdecode()
    205. 

  205.         {
    206. 

  206.             $char = $this->_getChar();
    207. 

  207.             switch ($char) {
    208. 

  208.             case 'i':
    209. 

  209.                 $this->_position++;
    210. 

  210.                 return $this->_decode_int();
    211. 

  211.                 break;
    212. 

  212.             case 'l':
    213. 

  213.                 $this->_position++;
    214. 

  214.                 return $this->_decode_list();
    215. 

  215.                 break;
    216. 

  216.             case 'd':
    217. 

  217.                 $this->_position++;
    218. 

  218.                 return $this->_decode_dict();
    219. 

  219.                 break;
    220. 

  220.             default:
    221. 

  221.                 return $this->_decode_string();
    222. 

  222.             }
    223. 

  223.         }

  225.         /**
    226. 

  226.         * Decode a BEncoded dictionary
    227. 

  227.         *
    228. 

  228.         * Dictionaries are prefixed with a d and terminated by an e. They
    229. 

  229.         * are similar to list, except that items are in key value pairs. The
    230. 

  230.         * dictionary {"key":"value", "Monduna":"com", "bit":"Torrents", "number":7}
    231. 

  231.         * would bEncode to d3:key5:value7:Monduna3:com3:bit:8:Torrents6:numberi7ee
    232. 

  232.         *
    233. 

  233.         * @access private
    234. 

  234.         * @return array 
    235. 

  235.         */
    236. 

  236.         function _decode_dict()
    237. 

  237.         {
    238. 

  238.             while ($this->_getChar() != 'e') {
    239. 

  239.                 $key = $this->_decode_string();
    240. 

  240.                 $val = $this->_bdecode();
    241. 

  241.                 $return[$key] = $val;
    242. 

  242.             }
    243. 

  243.             $this->_position++;
    244. 

  244.             return $return;
    245. 

  245.         }

  247.         /**
    248. 

  248.         * Decode a BEncoded string
    249. 

  249.         *
    250. 

  250.         * Strings are prefixed with their length followed by a colon.
    251. 

  251.         * For example, "Monduna" would bEncode to 7:Monduna and "BitTorrents"
    252. 

  252.         * would bEncode to 11:BitTorrents.
    253. 

  253.         *
    254. 

  254.         * @access private
    255. 

  255.         * @return string 
    256. 

  256.         */                
    257. 

  257.         function _decode_string()
    258. 

  258.         {
    259. 

  259.             // Find position of colon
    260. 

  260.             $pos_colon  = strpos($this->_source, ':', $this->_position);
    261. 

  261.             // Get length of string
    262. 

  262.             $str_length = intval(substr($this->_source, $this->_position, $pos_colon));
    263. 

  263.             // Get string
    264. 

  264.             $return = substr($this->_source, $pos_colon + 1, $str_length);
    265. 

  265.             // Move Pointer after string
    266. 

  266.             $this->_position = $pos_colon + $str_length + 1;
    267. 

  267.             return $return;
    268. 

  268.         }
    269.

  270.         /**
    271. 

  271.         * Decode a BEncoded integer
    272. 

  272.         *
    273. 

  273.         * Integers are prefixed with an i and terminated by an e. For
    274. 

  274.         * example, 123 would bEcode to i123e, -3272002 would bEncode to
    275. 

  275.         * i-3272002e.
    276. 

  276.         *
    277. 

  277.         * @access private
    278. 

  278.         * @return int 
    279. 

  279.         */                
    280. 

  280.         function _decode_int()
    281. 

  281.         {
    282. 

  282.             $pos_e  = strpos($this->_source, 'e', $this->_position);
    283. 

  283.             // $return = intval(substr($this->_source, $this->_position, $pos_e));
    284. 

  284.             $return = intval(substr($this->_source, $this->_position, $pos_e - $this->_position));
    285. 

  285.             $this->_position = $pos_e + 1;
    286. 

  286.             return $return;
    287. 

  287.         }

  289.         /**
    290. 

  290.         * Decode a BEncoded list
    291. 

  291.         *
    292. 

  292.         * Lists are prefixed with a l and terminated by an e. The list
    293. 

  293.         * should contain a series of bEncoded elements. For example, the
    294. 

  294.         * list of strings ["Monduna", "Bit", "Torrents"] would bEncode to
    295. 

  295.         * l7:Monduna3:Bit8:Torrentse. The list [1, "Monduna", 3, ["Sub", "List"]]
    296. 

  296.         * would bEncode to li1e7:Mondunai3el3:Sub4:Listee
    297. 

  297.         *
    298. 

  298.         * @access private
    299. 

  299.         * @return array 
    300. 

  300.         */                
    301. 

  301.         function _decode_list()
    302. 

  302.         {
    303. 

  303.             $char = $this->_getChar();
    304. 

  304.             while ($this->_source{$this->_position} != 'e') {
    305. 

  305.                 $val = $this->_bdecode();
    306. 

  306.                 $return[] = $val;
    307. 

  307.             }
    308. 

  308.             $this->_position++;
    309. 

  309.             return $return;
    310. 

  310.         }
