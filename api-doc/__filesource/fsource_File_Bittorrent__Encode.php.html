    28. 

  28. *
    29. 

  29. * BEncoding is a simple, easy to implement method of associating
    30. 

  30. * data types with information in a file. The values in a torrent
    31. 

  31. * file are bEncoded.
    32. 

  32. * There are 4 different data types that can be bEncoded:
    33.

  49. /**
    50. 

  50. * Encode files in Bittorrent format
    51. 

  51. *
    52. 

  52. * @package File_Bittorrent
    53. 

  53. * @category File
    54. 

  54. */
    55. 

  55. class File_Bittorrent_Encode
    56. 

  56. {
    57. 

  57.     /**
    58. 

  58.     * Encode a var in BEncode format
    59. 

  59.     *
    60. 

  60.     * @param mixed    Variable to encode
    61. 

  61.     * @return string 
    62. 

  62.     */                
    63. 

  63.     function encode($mixed)
    64. 

  64.     {
    65. 

  65.         switch (gettype($mixed)) {
    66. 

  66.         case 'string':
    67. 

  67.             return $this->encode_string($mixed);
    68. 

  68.             break;
    69. 

  69.         case 'integer':
    70. 

  70.             return  $this->encode_int($mixed);
    71. 

  71.             break;
    72. 

  72.         case 'array':
    73. 

  73.             return $this->encode_array($mixed);
    74. 

  74.             break;
    75. 

  75.         default:
    76. 

  76.             PEAR::raiseError('File_Bittorrent_Encode()::encode() - Unsupported type.', null, null, "Variable must be one of 'string', 'integer' or 'array'");
    77. 

  77.         }
    78. 

  78.     }
    79. 

  79.     
    80. 

  80.     /**
    81. 

  81.     * BEncodes a string
    82. 

  82.     *
    83. 

  83.     * Strings are prefixed with their length followed by a colon.
    84. 

  84.     * For example, "Monduna" would bEncode to 7:Monduna and "BitTorrents"
    85. 

  85.     * would bEncode to 11:BitTorrents.
    86. 

  86.     *
    87. 

  87.     * @param string 
    88. 

  88.     * @return string 
    89. 

  89.     */
    90. 

  90.     function encode_string($str)
    91. 

  91.     {
    92. 

  92.         return sprintf('%d:%s', strlen($str), $str);
    93. 

  93.     }
    94. 

  94.     
    95. 

  95.     /**
    96. 

  96.     * BEncodes a integer
    97. 

  97.     *
    98. 

  98.     * Integers are prefixed with an i and terminated by an e. For
    99. 

  99.     * example, 123 would bEcode to i123e, -3272002 would bEncode to
    100. 

  100.     * i-3272002e.
    101. 

  101.     *
    102. 

  102.     * @param int 
    103. 

  103.     * @return string 
    104. 

  104.     */
    105. 

  105.     function encode_int($int)
    106. 

  106.     {
    107. 

  107.         return sprintf('i%de', $int);
    108. 

  108.     }
    109. 

  109.     
    110. 

  110.     /**
    111. 

  111.     * BEncodes an array
    112. 

  112.     * This code assumes arrays with purely integer indexes are lists,
    113. 

  113.     * arrays which use string indexes assumed to be dictionaries.
    114. 

  114.     *
    115. 

  115.     * Dictionaries are prefixed with a d and terminated by an e. They
    116. 

  116.     * are similar to list, except that items are in key value pairs. The
    117. 

  117.     * dictionary {"key":"value", "Monduna":"com", "bit":"Torrents", "number":7}
    118. 

  118.     * would bEncode to d3:key5:value7:Monduna3:com3:bit:8:Torrents6:numberi7ee
    119. 

  119.     *
    120. 

  120.     * Lists are prefixed with a l and terminated by an e. The list
    121. 

  121.     * should contain a series of bEncoded elements. For example, the
    122. 

  122.     * list of strings ["Monduna", "Bit", "Torrents"] would bEncode to
    123. 

  123.     * l7:Monduna3:Bit8:Torrentse. The list [1, "Monduna", 3, ["Sub", "List"]]
    124. 

  124.     * would bEncode to li1e7:Mondunai3el3:Sub4:Listee
    125. 

  125.     *
    126. 

  126.     * @param array 
    127. 

  127.     * @return string 
    128. 

  128.     */
    129. 

  129.     function encode_array($array)
    130. 

  130.     {
    131. 

  131.         // Check for strings in the keys
    132. 

  132.         $isList = true;
    133. 

  133.         foreach (array_keys($array) as $key) {
    134. 

  134.             if (!is_int($key)) {
    135. 

  135.                 $isList = false;
    136. 

  136.                 break;
    137. 

  137.             }
    138. 

  138.         }
    139. 

  139.         if ($isList) {
    140. 

  140.             // Wie build a list
    141. 

  141.             $return = 'l';
    142. 

  142.             foreach ($array as $val) {
    143. 

  143.                 $return .= $this->encode($val);
    144. 

  144.             }
    145. 

  145.             $return .= 'e';
    146. 

  146.         } else {
    147. 

  147.             // We build a Dictionary
    148. 

  148.             $return = 'd';
    149. 

  149.             foreach ($array as $key => $val) {
    150. 

  150.                 $return .= $this->encode(strval($key));
    151. 

  151.                 $return .= $this->encode($val);
    152. 

  152.             }
    153. 

  153.             $return .= 'e';
    154. 

  154.         }
    155. 

  155.         return $return;
    156. 

  156.     }
    157. 

  157. }
